1. Install Java

rpm -aq | grep -i jdk yum remove mkdir -p /opt/java chmod -R 777 /opt/java cd /opt/java unzip jdk1.8.0_231.zip vim /etc/environment JAVA_HOME=/opt/java/jdk1.8.0_231 JRE_HOME=/opt/java/jdk1.8.0_231/jre

2. Install Tomcat (9091 & 9092)

sudo mkdir -p /opt/tomcat9091 sudo systemctl start tomcat9091 sudo systemctl enable tomcat9091 sudo firewall-cmd --zone=public --add-port=9091/tcp --permanent sudo firewall-cmd --reload

3. Create Tablespaces

CREATE TABLESPACE APEX DATAFILE 'apex1.dbf' SIZE 1G AUTOEXTEND ON; CREATE TABLESPACE APEX_FILE DATAFILE 'apex_file1.dbf' SIZE 1G AUTOEXTEND ON;

4. Install APEX

cd /home/oracle/apex sqlplus / as sysdba @apexins.sql APEX APEX_FILE TEMP /i/ @apxchpwd.sql

5. Install ORDS

cd /home/oracle mkdir ords cd ords unzip ords.zip java -jar medicare.war install cp medicare.war /opt/tomcat9091/webapps

6. Jasper Reports Configuration

url=jdbc:oracle:thin:@192.168.0.254:1521/msapxpdb username=medicare_new password=medicare_new

7. Run Application

Open browser: http://192.168.0.253:9091 http://192.168.0.253:9091/medicare

8. Create APEX Workspace

Workspace : MEDICARE Workspace ID : 1000001 User : admin Password : Po1$1wers

9. Application Configuration

REPORT_SERVER_IP = 192.168.0.253:9092
